## NOTE: This can take a while if there are a LOT of subfolders. If you see just running text, let it run.

robocopy C:\empty_tmp "C:\AI\SENNI-dev\.claude\worktrees\xenodochial-ritchie-d366fe\dist\senni-backend\backups" /MIR
rmdir "C:\AI\SENNI-dev\.claude\worktrees\xenodochial-ritchie-d366fe\dist\senni-backend\backups"
rmdir C:\empty_tmp
